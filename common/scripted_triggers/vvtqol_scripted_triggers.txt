# Province scope
# Arguments: 
vvtqol_trigger_can_construct_any_building = {
	# Check if it would be possible to construct a building manually
	
    NOT = { has_construction = missionary }
	NOT = { has_construction = culture }
	NOT = { has_construction = core }
	NOT = { has_construction = building }
    controlled_by = ROOT
    has_siege = no
    num_free_building_slots = 1
}

# Province scope
# Arguments: $BUILDING$ $BUILDING_COST$
vvtqol_trigger_can_construct_building = {
    vvtqol_trigger_can_construct_any_building = yes
    NOT = { has_building = $BUILDING$ }
    ROOT = {
        treasury = $BUILDING_COST$
    }
}

# Area scope

vvtqol_trigger_can_tc_invest = {

			investor = ROOT
			investment = $INVESTMENT$
		}
	}
	ROOT = {
		treasury = $INVESTMENT_COST$
	}
}

vvtqol_trigger_is_friendly_border_province = {
	AND = {	
		any_neighbor_province = {
			is_sea = no
			NOT = {
				owned_by = ROOT 
			}
			any_neighbor_province = {
				is_sea = no
				owned_by = ROOT	
			}
		}
		owned_by = ROOT
	}
}

vvtqol_trigger_is_any_province_allowed_for_automation = {
    any_owned_province = {
        limit = {
            vvtqol_trigger_is_province_allowed_for_automation = yes
            vvtqol_trigger_province_development_is_less_than_target_development = yes
        }
    }
}

vvtqol_trigger_province_development_is_less_than_target_development = {
    # PREV is here because the vvtqol_development_target is set in country scope and vvtqol_province_development is set in province scope.
    PREV = {
        variable_arithmetic_trigger = {
            export_to_variable = {
                which = vvtqol_province_development
                value = development
                who = PREV
            }
            
            # Check if province development is less than some number specified by a variable.
            NOT = {
                check_variable = {
                    which = vvtqol_province_development 
                    which = vvtqol_development_target   
                }
            }
        }
    }
}
